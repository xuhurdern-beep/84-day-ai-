try:
    # 【可能出错的代码】：比如文件读写、类型转换
    可能报错的代码块
except 异常类型1 as e:
    # 【捕获到异常1时执行】：处理对应错误
    错误处理代码1
except 异常类型2 as e:
    # 【捕获到异常2时执行】
    错误处理代码2
except Exception as e:
    # 【兜底：捕获所有其他未处理的异常】
    通用错误处理代码
else:
    # 【无任何异常时执行】：可选，放正常逻辑
    正常执行代码
finally:
    # 【无论是否异常，100%执行】：可选，用于资源释放（with已自动处理）
    必执行代码
